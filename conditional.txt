.model small
.stack 100h
.data
msg db 'Enter the number : $'
msg1 db 'The number is Greater than 5 $'
msg2 db 'The number is Equal to 5 $'
msg3 db 'The number is Less than 5 $'
.code
main proc
	mov ax,@data
	mov ds,ax
	
	mov ah,9
	lea dx,msg
	int 21h

	mov ah,1
	int 21h

	mov bl,al

	mov ah,2
	mov dl,0ah
	int 21h
	mov dl,0dh
	int 21h

	cmp bl,'5'
	jg GG
	
	cmp bl,'5'
	je EE

	cmp bl,'5'
	jl LL

	jmp exit

GG:
	mov ah,9
	lea dx,msg1
	int 21h

	jmp exit
EE:
	mov ah,9
	lea dx,msg2
	int 21h

	jmp exit
LL:
	mov ah,9
	lea dx,msg3
	int 21h

	jmp exit
exit:

    main endp
end main
