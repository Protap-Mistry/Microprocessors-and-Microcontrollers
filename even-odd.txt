.model small
.stack 100h
.data
msg db 'Enter the number : $'
msg1 db 'The number is EVEN $'
msg2 db 'The number is ODD $'
.code
main proc
	mov ax,@data
	mov ds,ax
	
	mov ah,9
	lea dx,msg
	int 21h

	mov ah,1
	int 21h
	
	mov cl,2
	div cl

	mov bl,ah

	mov ah,2
	mov dl,0ah
	int 21h
	mov dl,0dh
	int 21h

	cmp bl,0
	je even
	
	cmp bl,0
	jne odd	

	jmp exit

even:
	mov ah,9
	lea dx,msg1
	int 21h

	jmp exit
odd:
	mov ah,9
	lea dx,msg2
	int 21h

	jmp exit

exit:

    main endp
end main
