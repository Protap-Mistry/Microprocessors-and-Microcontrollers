.model small
.stack 100h

.data
.code
main proc
    mov cx,0
    mov ah,1
    
    read:
        int 21h
        cmp al,0dh
        je reverse
        push ax
        inc cx
        jmp read
        
    reverse:
        cmp cx,0
        je exit
        mov ah,2
        mov dl,0ah
        int 21h
        mov dl,0dh
        int 21h
        mov ah,2
        
    print:
        pop dx
        int 21h
        loop print
        
    exit:
    
    main endp

end main
